\begin{algorithm}[H]
    \caption{Algo3}\label{alg:pruning}
    \begin{algorithmic}[1]
        \STATE{\textbf{INPUT:} $X,\mathcal{S}$ of size $n,m$ and $t \in [1..n]$}\\
        \STATE{\textbf{RESULT:} $\mathcal{P}  = \{P_1,...,P_t\}$ with $P_*$ of size $n/t$}\\
        \STATE{$\forall S \in \mathcal{S}, \pi(S) \leftarrow 1$}
        \STATE{$\mathcal{P} \leftarrow \emptyset$}
        \FOR{$i \leftarrow 1$ to $t$}
            \STATE{$P_i \leftarrow \emptyset$}
            \STATE{$E \leftarrow \text{pairs}(X)$}
            \FOR{$n/t - 1$ steps}
                \STATE{$\forall e \in E, \omega(e) \leftarrow \sum\limits_{S \in \mathcal{S}}\pi(S)I(e,S)$}
                \IF{$|P_i| = 0$}
                    \STATE{$(x_j,x_k) \leftarrow$ arbitrary edge of minimum weight}
                    \STATE{$P_i \leftarrow P_i \cup \{x_k,x_j\}$}
                \ELSE
                    \STATE{$(x_j,x_k) \leftarrow$ arbitrary edge of minimum weight with $x_j \in P_i$ \textcolor{orange}{(or $x_k \in P_i$)}}
                    \STATE{$P_i \leftarrow P_i \cup \{x_k\}$ \textcolor{orange}{($P_i \leftarrow P_i \cup \{x_j\}$)}}
                \ENDIF
                \STATE{$\forall (u,v)\in P_i^2$ remove $(u,v)$ from $E$ \textcolor{blue}{\#Remove edges between points in the partition}}
                \STATE{$\forall s \in \mathcal{S}, \pi(S) = \pi(S) - \pi(S)I((x_j,x_k),S)$ \textcolor{blue}{\#Set weight of sets crossing the selected edge to 0}}
            \ENDFOR
            \STATE{$\mathcal{P} \leftarrow \mathcal{P} \cup P_i$}
            \STATE{$X \leftarrow X \backslash P_i$}
            \STATE{$\forall S \in \mathcal{S}, \pi(S) = 2^{\sum\limits_{k=1}^i I(P_i,S)}$}
        \ENDFOR
		\RETURN{$\mathcal{P}$}
	\end{algorithmic}
\end{algorithm}
